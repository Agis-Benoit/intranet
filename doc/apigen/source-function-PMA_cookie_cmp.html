292:             $PHP_AUTH_USER = $pma_username;

296:

306:                 $PHP_AUTH_USER = $_COOKIE['pma_cookie_username'];
307:                 $from_cookie   = TRUE;
308:             }
309:             else if (!empty($HTTP_COOKIE_VARS) && isset($HTTP_COOKIE_VARS['pma_cookie_username'])) {
310:                 $PHP_AUTH_USER = $HTTP_COOKIE_VARS['pma_cookie_username'];
